import cv2

import albumentations as A

import os

from pathlib import Path

from tqdm import tqdm

import shutil

 

#PATHS

input_root = Path("ClearWaterv1")  # Your current folder

output_root = Path("PondWaterv1") # The new filtered folder

 

# subfolders,

subsets = ['train', 'valid']

 

#pond filter

pond_transform = A.Compose([

    #Murky Green/Brown Tint

    A.RGBShift(r_shift_limit=(-0.12, -0.12), g_shift_limit=(0.08, 0.08), b_shift_limit=(-0.16, -0.16), p=1.0),

   

    #Blur

    A.MedianBlur(blur_limit=5, p=0.7),

   

    #Adding a very fine texture so the water isn't "perfectly" flat

    A.GaussNoise(var_limit=(0.005, 0.02), p=0.5),

 

    #brightness and contrast

    A.RandomBrightnessContrast(brightness_limit=-0.2, contrast_limit=-0.3, p=1.0),

])

 

def process_dataset():

    for subset in subsets:

        print(f"\nProcessing {subset} set...")

       

        # Define paths for this subset

        img_input_dir = input_root / subset / "images"

        lbl_input_dir = input_root / subset / "labels"

       

        img_output_dir = output_root / subset / "images"

        lbl_output_dir = output_root / subset / "labels"

       

        # Create output folders

        img_output_dir.mkdir(parents=True, exist_ok=True)

        lbl_output_dir.mkdir(parents=True, exist_ok=True)

       

        # Get all images

        image_paths = list(img_input_dir.glob("*.jpg")) + list(img_input_dir.glob("*.png"))

       

        for img_path in tqdm(image_paths):

            # Process Image

            image = cv2.imread(str(img_path))

            if image is None: continue

            image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)

           

            transformed = pond_transform(image=image)["image"]

            transformed_bgr = cv2.cvtColor(transformed, cv2.COLOR_RGB2BGR)

           

            # Save Image

            cv2.imwrite(str(img_output_dir / img_path.name), transformed_bgr)

           

            #  Copy Label (if exists)

            label_path = lbl_input_dir / f"{img_path.stem}.txt"

            if label_path.exists():

                shutil.copy(label_path, lbl_output_dir / label_path.name)

 

    # Copy Metadata Files

    for meta_file in ['data.yaml', 'README.dataset.txt', 'README.roboflow.txt']:

        if (input_root / meta_file).exists():

            shutil.copy(input_root / meta_file, output_root / meta_file)

 

if __name__ == "__main__":

    process_dataset()

    print(f"\nSuccessfully created {output_root}")
